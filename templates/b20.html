B20 tokens are deployed via Base's native B20Factory precompile. This requires base-forge from the terminal — not MetaMask. Follow the steps below:

Step 1 — Install base-forge

curl -L https://raw.githubusercontent.com/base-org/base-forge/main/install.sh | bash
source ~/.bashrc
base-forge --version

Step 2 — Check activation status

REG=0x8453000000000000000000000000000000000001
RPC=https://mainnet.base.org
base-cast call $REG "isActivated(bytes32)(bool)" \
  $(base-cast keccak "base.b20_asset") --rpc-url $RPC

⚠️ Activation Registry goes live July 8, 2026 at 18:00 UTC

Step 3 — Create deploy script

Create script/CreateToken.s.sol:

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;
import {Script} from "forge-std/Script.sol";
import {B20FactoryLib} from "base-std/lib/B20FactoryLib.sol";
import {IB20Factory} from "base-std/interfaces/IB20Factory.sol";
import {StdPrecompiles} from "base-std/StdPrecompiles.sol";

contract CreateToken is Script {
  function run() external returns (address token) {
    vm.startBroadcast();
    bytes32 salt = keccak256("my-token-v1");
    bytes memory params = B20FactoryLib.encodeAssetParams(
      "My Token", "MTK", msg.sender, 18
    );
    token = IB20Factory(StdPrecompiles.B20_FACTORY)
      .createB20("ASSET", salt, params, new bytes[](0));
    vm.stopBroadcast();
  }
}

Step 4 — Deploy

base-forge script script/CreateToken.s.sol \
  --rpc-url https://mainnet.base.org \
  --broadcast \
  --private-key YOUR_PRIVATE_KEY

⚠️ Never share your private key. Use a .env file.
