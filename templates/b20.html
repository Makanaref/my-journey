async function deployB20() { const name = document.getElementById("deploy-name").value.trim(); const symbol = document.getElementById("deploy-symbol").value.trim(); const decimals = parseInt(document.getElementById("deploy-decimals").value); const cap = document.getElementById("deploy-cap").value.trim(); const mintRole = document.getElementById("deploy-mint-role").checked; const statusEl = document.getElementById("deploy-status"); const currency = document.getElementById("deploy-currency")?.value.trim() || "USD"; // --- Validation --- if (!name || !symbol) { statusEl.innerText = "Please fill name and symbol."; return; } if (deployType === 'stable') { if (decimals !== 6) { statusEl.innerText = "Stablecoin decimals must be 6."; return; } if (!currency || currency.length !== 3) { statusEl.innerText = "Please enter a valid 3-letter ISO currency code (e.g. USD)."; return; } } else { if (decimals < 6 || decimals > 18) { statusEl.innerText = "Decimals must be between 6 and 18."; return; } } statusEl.innerText = "Preparing transaction..."; document.getElementById("deploy-btn").disabled = true; try { // --- Core Deployment Logic --- // 1. Correct Factory Address (with capital F) const B20_FACTORY = "0xB20F000000000000000000000000000000000000"; // 2. Variant and Salt const variant = deployType === 'stable' ? 1 : 0; const salt = ethers.keccak256(ethers.toUtf8Bytes(name + symbol + Date.now())); // 3. Build Params (using tuple with version byte, as per working examples) const abiCoder = ethers.AbiCoder.defaultAbiCoder(); let params; if (deployType === 'stable') { const paramsStruct = [1, name, symbol, account, currency.toUpperCase().slice(0, 3)]; params = abiCoder.encode(["tuple(uint8,string,string,address,string)"], [paramsStruct]); } else { const paramsStruct = [1, name, symbol, account, decimals]; params = abiCoder.encode(["tuple(uint8,string,string,address,uint8)"], [paramsStruct]); } // 4. Build InitCalls (optimized: only add if needed) const initCalls = []; const grantRoleInterface = new ethers.Interface(["function grantRole(bytes32 role, address account)"]); const capInterface = new ethers.Interface(["function updateSupplyCap(uint256 cap)"]); if (mintRole) { const MINT_ROLE = ethers.keccak256(ethers.toUtf8Bytes("MINT_ROLE")); initCalls.push(grantRoleInterface.encodeFunctionData("grantRole", [MINT_ROLE, account])); } if (cap && Number(cap) > 0) { const capValue = ethers.parseUnits(cap, decimals); initCalls.push(capInterface.encodeFunctionData("updateSupplyCap", [capValue])); } // 5. Encode Factory Call const factoryInterface = new ethers.Interface([ "function createB20(uint8 variant, bytes32 salt, bytes params, bytes[] initCalls) returns (address)" ]); const data = factoryInterface.encodeFunctionData("createB20", [variant, salt, params, initCalls]); // --- Gas Optimization --- // 1. Estimate Gas const estimatedGas = await provider.estimateGas({ to: B20_FACTORY, data: data, from: account }); // 2. Use 90% of suggested gas price const feeData = await provider.getFeeData(); const maxGasPrice = feeData.gasPrice * 90n / 100n; // 3. Use 105% of estimated gas as limit (5% safety margin) const gasLimit = estimatedGas * 105n / 100n; console.log("Gas optimization:", { estimatedGas: estimatedGas.toString(), gasLimit: gasLimit.toString(), maxGasPrice: maxGasPrice.toString(), originalPrice: feeData.gasPrice.toString() }); // 4. Send Transaction const tx = await signer.sendTransaction({ to: B20_FACTORY, data: data, gasPrice: maxGasPrice, gasLimit: gasLimit }); statusEl.innerText = "Confirming transaction..."; const receipt = await tx.wait(); // --- Extract Token Address --- // The token address is in the logs and should start with 0xB20... const tokenAddress = receipt.logs.find(log => log.address.toLowerCase().startsWith("0xb20") )?.address || "Check explorer"; // --- Update UI --- document.getElementById("deploy-address").innerText = tokenAddress; document.getElementById("deploy-result").style.display = "block"; statusEl.innerText = "Deployed successfully! ✅"; } catch (err) { console.error("Full error:", err); let errorMsg = err.message; if (err.data) { try { if (err.data.startsWith("0x")) { const hexStr = err.data.slice(2); const bytes = new Uint8Array(hexStr.match(/.{1,2}/g).map(byte => parseInt(byte, 16))); const decoded = new TextDecoder().decode(bytes); if (decoded) errorMsg = decoded; } } catch { errorMsg = err.data; } } statusEl.innerText = "Failed: " + (err.reason || errorMsg || "Unknown error"); } document.getElementById("deploy-btn").disabled = false; }
